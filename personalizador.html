function processAll(img) { if (Math.abs(img.width - TARGET_W) > 100) { updateCheck('c-size', 'fail', `Tamanho errado: ${img.width}px`); return; } updateCheck('c-size', 'ok', 'Tamanho OK (25x5cm)'); canvas.width = TARGET_W; canvas.height = TARGET_H; ctx.fillStyle = "white"; ctx.fillRect(0, 0, TARGET_W, TARGET_H); ctx.drawImage(img, 0, 0, TARGET_W, TARGET_H); let imgData = ctx.getImageData(0, 0, TARGET_W, TARGET_H); let data = imgData.data; // 1. Converter para P&B Puro for (let i = 0; i < data.length; i += 4) { let avg = (data[i] + data[i+1] + data[i+2]) / 3; let v = avg < 200 ? 0 : 255; data[i] = data[i+1] = data[i+2] = v; } updateCheck('c-color', 'ok', 'Cores OK'); // 2. Criar Canvas de Preview para desenhar os erros let errorCanvas = document.createElement('canvas'); errorCanvas.width = TARGET_W; errorCanvas.height = TARGET_H; let eCtx = errorCanvas.getContext('2d'); eCtx.putImageData(imgData, 0, 0); // Começa com a imagem P&B limpa eCtx.fillStyle = "rgba(210, 59, 64, 0.8)"; // Vermelho Coisoteca com leve transparência let errorCount = 0; // 3. SCANNER HORIZONTAL (Detecta linhas verticais finas) for (let y = 0; y < TARGET_H; y += 3) { // Salto de 3px para não pesar tanto, mas ler tudo let streak = 0; for (let x = 0; x < TARGET_W; x++) { if (data[(y * TARGET_W + x) * 4] === 0) { streak++; } else { if (streak > 0 && streak < MIN_LINE_PX) { // Marca o erro no canvas de preview eCtx.fillRect(x - streak, y - 5, streak, 10); errorCount++; } streak = 0; } } } // 4. SCANNER VERTICAL (Detecta linhas horizontais finas) for (let x = 0; x < TARGET_W; x += 10) { let streak = 0; for (let y = 0; y < TARGET_H; y++) { if (data[(y * TARGET_W + x) * 4] === 0) { streak++; } else { if (streak > 0 && streak < MIN_LINE_PX) { eCtx.fillRect(x - 5, y - streak, 10, streak); errorCount++; } streak = 0; } } } if (errorCount > 0) { updateCheck('c-thick', 'fail', `${errorCount} pontos críticos detectados (<1mm)`); previewMsg.innerText = "⚠️ Corrija os pontos vermelhos para imprimir"; previewMsg.style.color = "var(--red)"; } else { updateCheck('c-thick', 'ok', 'Espessura aprovada em toda a arte'); previewMsg.innerText = "✅ Arte perfeita para impressão"; previewMsg.style.color = "var(--green)"; } // Atualiza Preview e Botão previewImg.src = errorCanvas.toDataURL("image/png"); previewContainer.style.display = 'flex'; btnDown.style.display = 'block'; btnDown.onclick = () => { ctx.putImageData(imgData, 0, 0); // Download sempre limpo const link = document.createElement('a'); link.download = 'placa_coisoteca.png'; link.href = canvas.toDataURL("image/png"); link.click(); }; }
