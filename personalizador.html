function processAll(img) { if (Math.abs(img.width - TARGET_W) > 100) { updateCheck('c-size', 'fail', `Tamanho errado: ${img.width}px`); return; } updateCheck('c-size', 'ok', 'Tamanho OK (25x5cm)'); canvas.width = TARGET_W; canvas.height = TARGET_H; ctx.fillStyle = "white"; ctx.fillRect(0, 0, TARGET_W, TARGET_H); ctx.drawImage(img, 0, 0, TARGET_W, TARGET_H); let imgData = ctx.getImageData(0, 0, TARGET_W, TARGET_H); let data = imgData.data; // 1. Converter para P&B Puro for (let i = 0; i < data.length; i += 4) { let avg = (data[i] + data[i+1] + data[i+2]) / 3; let v = avg < 200 ? 0 : 255; data[i] = data[i+1] = data[i+2] = v; } updateCheck('c-color', 'ok', 'Cores OK'); // 2. Canvas de Preview let errorCanvas = document.createElement('canvas'); errorCanvas.width = TARGET_W; errorCanvas.height = TARGET_H; let eCtx = errorCanvas.getContext('2d'); eCtx.putImageData(imgData, 0, 0); eCtx.fillStyle = "rgba(210, 59, 64, 0.9)"; // Vermelho mais opaco let hasError = false; let errorCount = 0; // 3. SCANNER HORIZONTAL (Detecta larguras finas - Ex: Pontos, Linhas Verticais) for (let y = 0; y < TARGET_H; y += 2) { let streak = 0; for (let x = 0; x < TARGET_W; x++) { if (data[(y * TARGET_W + x) * 4] === 0) { streak++; } else { if (streak > 0 && streak < MIN_LINE_PX) { eCtx.fillRect(x - streak, y - 4, streak, 8); errorCount++; hasError = true; } streak = 0; } } } // 4. SCANNER VERTICAL (Detecta alturas finas - Ex: A sua linha comprida horizontal) // Agora ele varre pixel por pixel lateralmente também for (let x = 0; x < TARGET_W; x += 2) { let streak = 0; for (let y = 0; y < TARGET_H; y++) { if (data[(y * TARGET_W + x) * 4] === 0) { streak++; } else { if (streak > 0 && streak < MIN_LINE_PX) { // Marca o erro da linha horizontal fina eCtx.fillRect(x - 4, y - streak, 8, streak); errorCount++; hasError = true; } streak = 0; } } } if (hasError) { updateCheck('c-thick', 'fail', `Erros detectados: ajuste os traços vermelhos.`); previewMsg.innerText = "⚠️ Linhas finas detectadas (vermelho)"; previewMsg.style.color = "var(--red)"; } else { updateCheck('c-thick', 'ok', 'Espessura aprovada!'); previewMsg.innerText = "✅ Arte pronta para impressão"; previewMsg.style.color = "var(--green)"; } previewImg.src = errorCanvas.toDataURL("image/png"); previewContainer.style.display = 'flex'; btnDown.style.display = 'block'; btnDown.onclick = () => { ctx.putImageData(imgData, 0, 0); const link = document.createElement('a'); link.download = 'placa_coisoteca.png'; link.href = canvas.toDataURL("image/png"); link.click(); }; }
